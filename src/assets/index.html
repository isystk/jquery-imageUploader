使い方

Javascript

<script src="https://demo.isystk.com/jquery-zoomSlider/js/jquery-imageUploader.js"></script>
<script type="text/javascript">
$(function() {
	$('.js-uploadImage').imageUploader({
		dropAreaSelector: '#drop-zone',
		successCallback: function(res) {

			$([
				'<tr>',
					'<td><img src="'+res.fileData+'" /></td>',
					'<td>'+res.fileName+'</td>',
					'<td>'+res.fileSize+'</td>',
					'<td>'+res.fileType+'</td>',
				'</tr>'
			].join('')).appendTo('#select-image table');
		}
	});
});
</script>
